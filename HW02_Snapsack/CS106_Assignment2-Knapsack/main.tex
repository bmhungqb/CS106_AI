\section{Đặc trưng bài toán Snapsack}
\begin{itemize}
    \item Hàm mục tiêu: Max tổng giá trị các đồ vật của túi
    \item Tổng các đồ vật trong túi phụ thuộc vào 2 yếu tố
        \begin{itemize}
            \item Tổng khối lượng các đồ vật có trong túi
            \item Tổng giá trị các đồ vật trong túi
        \end{itemize}
\end{itemize}

        \end{itemize}
    \item Time limit: 3 phút
\end{itemize}
